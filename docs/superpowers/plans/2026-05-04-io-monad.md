# IO Monad Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a built-in monomorphic `IO` monad with `pure`, `bind`, `print`, `readNat`, plus a `Unit` type, so the program `main = bind readNat (\n. print (mul n 2))` reads a number and prints its double.

**Architecture:** `IO a` is a primitive opaque type constructor; the four IO primitives build an `IOAction` operation tree at WHNF time (no side effects yet). A `run_io` driver walks the tree at the top level, performing stdin/stdout. File-mode `main` must have type `IO a`; REPL evaluates IO if it sees one and pretty-prints otherwise.

**Tech Stack:** Rust 2021, chumsky parser combinators (already used), existing CBN evaluator with thunks.

**Spec:** `docs/superpowers/specs/2026-05-04-io-monad-design.md`

pub fn unify(a: &Type, b: &Type) -> Result<Subst, TypeError> {
    match (a, b) {
        (Type::Var(x), Type::Var(y)) if x == y => Ok(Subst::empty()),
        (Type::Nat, Type::Nat) => Ok(Subst::empty()),
        (Type::Unit, Type::Unit) => Ok(Subst::empty()),
        (Type::Var(x), t) | (t, Type::Var(x)) => bind(*x, t),
        (Type::Arrow(a1, b1), Type::Arrow(a2, b2)) => {
            let s1 = unify(a1, a2)?;
            let s2 = unify(&s1.apply(b1), &s1.apply(b2))?;
            Ok(s2.compose(&s1))
        }
        (Type::IO(a), Type::IO(b)) => unify(a, b),
        _ => Err(TypeError::Mismatch(a.clone(), b.clone())),
    }
}
```

    // Unit), then if any args remain the user has applied `()` to
    // something — surface as stuck.
    loop {
        match stack.pop() {
            Some(Frame::Update(node)) => {
                *node.thunk.borrow_mut() = Thunk::ForcedUnit;
            }
            Some(Frame::Arg(t, e)) | Some(Frame::StrictArg(t, e)) => {
                let mut args = vec![(t, e)];
                while let Some(fr) = stack.pop() {
                    match fr {
                        Frame::Arg(t, e) | Frame::StrictArg(t, e) => args.push((t, e)),
                        Frame::Update(_) => {}
                    }
                }
                return Ok(Value::StuckApp {
                    head: DBExpr::UnitLit,
                    args,
                });
            }
            None => return Ok(Value::Unit),
        }
    }
}

                    Frame::Arg(t, e) | Frame::StrictArg(t, e) => args.push((t, e)),

            try_force_nat(&popped[1].0, &popped[1].1, budget)?,
        ) {
            (Some(a), Some(b)) => Some((DBExpr::NatLit(a.saturating_sub(b)), empty_env())),
            _ => None,

        Mul => match (
            try_force_nat(&popped[0].0, &popped[0].1, budget)?,
            try_force_nat(&popped[1].0, &popped[1].1, budget)?,
        ) {
            (Some(a), Some(b)) => Some((DBExpr::NatLit(a.saturating_mul(b)), empty_env())),
            _ => None,
        },
        IfZ => match try_force_nat(&popped[0].0, &popped[0].1, budget)? {
            Some(0) => Some(popped[1].clone()),
            Some(_) => Some(popped[2].clone()),
            None => None,
        },
        Pure | Bind | Print | ReadNat => unreachable!("handled above"),
    };
    match result {
        Some((next_focus, next_env)) => {
            focus = next_focus;
            env = next_env;
        }
        None => {
            return Ok(Value::StuckApp {
                head: DBExpr::Prim(op),
                args: popped,
            });
        }
    }
}

/// `nf`. This is the inverse of the WHNF rules: `Pure(t,e)` becomes
/// `Prim(Pure) (nf t e)`, `Bind(a, k_t, k_e)` becomes

/// Used only for pretty-printing IO values that leak out of `whnf` into a
/// non-IO context. The runtime driver consumes IOActions before they reach
/// pretty-printing in normal program flow.

    a: &IOAction,
    depth: usize,
    budget: &mut Budget,

    use crate::ast::PrimOp::*;
    match a {

            let inner = nf(t, e, depth, budget)?;

        IOAction::Print(t, e) => {

            Ok(DBExpr::app(DBExpr::Prim(Print), inner))

            let kont = nf(k_t, k_e, depth, budget)?;

                DBExpr::app(DBExpr::Prim(Bind), inner),

        }
    }

//! Runtime driver for `IO` actions. Walks an `IOAction` tree built by
//! WHNF and performs the actual side effects via injectable `IoSource`
//! and `IoSink` traits.
//!
//! The driver is iterative (loop, not recursion) so deep `bind` chains
//! don't blow the stack.

use std::rc::Rc;

use crate::cbn::{empty_env, whnf, Budget, IOAction, Thunk, Value};
use crate::debruijn::DBExpr;
use crate::error::EvalError;

/// Source of input for `readNat`. Real programs use `StdinSource`; tests
/// substitute a `Vec<String>`-backed mock.
pub trait IoSource {
    /// Return the next input line (without trailing newline), or `None`
    /// at EOF.
    fn read_line(&self) -> Option<String>;
}

/// Output sink for `print`. Real programs use `StdoutSink`; tests
/// substitute a `String`-backed mock.
pub trait IoSink {
    fn writeln_nat(&self, n: u64);
}

/// Production stdin source.
pub struct StdinSource;

impl IoSource for StdinSource {
    fn read_line(&self) -> Option<String> {
        let mut buf = String::new();
        match std::io::stdin().read_line(&mut buf) {
            Ok(0) => None,
            Ok(_) => Some(buf.trim_end_matches('\n').trim_end_matches('\r').to_string()),
            Err(_) => None,
        }
    }
}

/// Production stdout sink.
pub struct StdoutSink;

impl IoSink for StdoutSink {
    fn writeln_nat(&self, n: u64) {
        println!("{}", n);
    }
}

/// Force a `(term, env)` thunk to a `Value::Nat`. Returns `Err` if the
/// term reduces to anything other than a `Nat`. Used by `Print` and the
/// final-result step of the driver.
fn force_to_nat(
    term: &DBExpr,
    env: &crate::cbn::Env,
    budget: &mut Budget,
) -> Result<u64, EvalError> {
    let v = whnf(term, env, budget)?;
    match v {
        Value::Nat(n) => Ok(n),
        other => Err(EvalError::Runtime(format!(
            "runtime error: expected Nat, got {:?}",
            other
        ))),
    }
}

/// Force a `(term, env)` thunk to a `Value` (any kind). Used by `Pure`
/// to yield its argument.
fn force_value(
    term: &DBExpr,
    env: &crate::cbn::Env,
    budget: &mut Budget,
) -> Result<Value, EvalError> {
    whnf(term, env, budget)
}

/// Force a `(term, env)` thunk and require it be a `Value::IOAction`.
/// Used inside `Bind` after applying the continuation.
fn force_to_ioaction(
    term: &DBExpr,
    env: &crate::cbn::Env,
    budget: &mut Budget,
) -> Result<Rc<IOAction>, EvalError> {
    let v = whnf(term, env, budget)?;
    match v {
        Value::IOAction(a) => Ok(a),
        other => Err(EvalError::Runtime(format!(
            "runtime error: expected IO action from continuation, got {:?}",
            other
        ))),
    }
}

/// Apply a continuation `(k_term, k_env) : a -> IO b` to the runtime
/// `Value` produced by the previous action.
///
/// Mechanism: WHNF the continuation to a `Closure`; then push the value
/// into the closure's env as a `Forced*` thunk and WHNF the body — which
/// yields a `Value::IOAction`.
fn apply_continuation(
    k_term: &DBExpr,
    k_env: &crate::cbn::Env,
    arg: Value,
    budget: &mut Budget,
) -> Result<Rc<IOAction>, EvalError> {
    use std::cell::RefCell;

    // Reduce the continuation to a closure.
    let k_val = whnf(k_term, k_env, budget)?;
    let closure = match k_val {
        Value::Cls(c) => c,
        other => {
            return Err(EvalError::Runtime(format!(
                "runtime error: continuation is not a function: {:?}",
                other
            )));
        }
    };

    // Build a Forced* thunk for the arg and push onto the closure's env.
    let thunk = match arg {
        Value::Nat(n) => Thunk::ForcedNat(n),
        Value::Unit => Thunk::ForcedUnit,
        Value::IOAction(a) => Thunk::ForcedIOAction(a),
        Value::Cls(c) => Thunk::Forced(c),
        Value::Neu { .. } | Value::StuckApp { .. } => {
            return Err(EvalError::Runtime(
                "runtime error: continuation arg is a stuck/neutral value".into(),
            ));
        }
    };
    let new_node = Rc::new(crate::cbn::EnvNode {
        thunk: RefCell::new(thunk),
        tail: closure.env.clone(),
    });

    // WHNF the body in the extended env. Should yield a Value::IOAction.
    let body_v = whnf(&closure.body, &Some(new_node), budget)?;
    match body_v {
        Value::IOAction(a) => Ok(a),
        other => Err(EvalError::Runtime(format!(
            "runtime error: continuation body is not an IO action: {:?}",
            other
        ))),
    }
}

/// Drive an `IOAction` tree, performing side effects. Iterative — a
/// `Bind(a, k)` walks `a` first (recursively, but bounded by the inner-
/// action depth), then re-enters the loop with the continuation's
/// resulting action. Right-nested `bind` chains stay flat.
pub fn run_io(
    action: &Rc<IOAction>,
    source: &dyn IoSource,
    sink: Rc<dyn IoSink>,
    budget: &mut Budget,
) -> Result<Value, EvalError> {
    let mut current: Rc<IOAction> = Rc::clone(action);
    loop {
        match &*current.clone() {
            IOAction::Pure(t, e) => return force_value(t, e, budget),
            IOAction::Print(t, e) => {
                let n = force_to_nat(t, e, budget)?;
                sink.writeln_nat(n);
                return Ok(Value::Unit);
            }
            IOAction::ReadNat => {
                let line = source.read_line().ok_or_else(|| {
                    EvalError::Runtime(
                        "runtime error: readNat: end of input".into(),
                    )
                })?;
                let n: u64 = line.trim().parse().map_err(|_| {
                    EvalError::Runtime(format!(
                        "runtime error: readNat: could not parse '{}' as Nat",
                        line.trim()
                    ))
                })?;
                return Ok(Value::Nat(n));
            }
            IOAction::Bind(inner, k_t, k_e) => {
                let inner_val = run_io(inner, source, Rc::clone(&sink), budget)?;
                let next = apply_continuation(k_t, k_e, inner_val, budget)?;
                current = next;
                // Loop iterates with the new action.
            }
        }
    }
}

            eprintln!(": {}", s);
            Some(t.clone())
        }
        Some(Err(e)) => {
            eprintln!(": (type error: {})", e);
            had_type_error = true;
            None
        }
        None => None,
    };

    if had_type_error && !no_typecheck {
        eprintln!("aborting: type errors above (re-run with --no-typecheck to evaluate anyway)");
        process::exit(1);
    }

    if program.main.is_none() {
        for d in &program.defs {
            println!("def {} = {}", d.name, print(&d.body));
        }
        return;
    }

    // Strict main:IO a check — only when typecheck is on.
    let main_is_io = if let Some(t) = &main_ty {
        let mut fresh = lc::infer::Fresh::new();
        let alpha = fresh.tvar();
        lc::types::unify(t, &Type::IO(Box::new(alpha))).is_ok()
    } else {
        false
    };

    if !no_typecheck {
        if let Some(t) = &main_ty {
            if !main_is_io {
                eprintln!(
                    "aborting: main must have type 'IO a', got '{}'",
                    t,
                );
                process::exit(1);
            }
        }
    }

    let inlined = match inline_defs(&program) {
        Ok(e) => e,
        Err(e) => {
            eprintln!("{}", e);
            process::exit(1);
        }
    };
    let prepared = if no_simplify { inlined } else { simplify(&inlined) };

    if main_is_io {
        // IO mode: WHNF (not full NF), then drive the action.
        let db = to_db(&prepared);
        let db = if no_strict_eval { db } else { mark_strict(&db) };
        let mut budget = Budget::new(DEFAULT_STEP_LIMIT);
        let v = match cbn::whnf(&db, &cbn::empty_env(), &mut budget) {
            Ok(v) => v,
            Err(e) => {
                eprintln!("{}", e);
                process::exit(1);
            }
        };
        let action = match v {
            cbn::Value::IOAction(a) => a,
            other => {
                eprintln!(
                    "internal error: main typechecked as IO but evaluated to {:?}",
                    other,
                );
                process::exit(1);
            }
        };
        let source = StdinSource;
        let sink: std::rc::Rc<dyn lc::io_runtime::IoSink> = std::rc::Rc::new(StdoutSink);
        match run_io(&action, &source, sink, &mut budget) {
            Ok(cbn::Value::Unit) => {} // nothing to print
            Ok(cbn::Value::Nat(n)) => println!("{}", n),
            Ok(other) => println!("{:?}", other),
            Err(e) => {
                eprintln!("{}", e);
                process::exit(1);
            }
        }
    } else {
        // Non-IO mode (only reachable with --no-typecheck): full NF + pretty-print.
        match normalize_with_options(&prepared, DEFAULT_STEP_LIMIT, !no_strict_eval) {
            Ok((nf, _)) => println!("{}", print(&nf)),
            Err(e) => {
                eprintln!("{}", e);
                process::exit(1);
            }
        }
    }
}
```
